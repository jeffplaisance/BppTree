35 template <typename T = Empty, bool enable_exclude = !std::is_same_v<T, Empty>>
36 IndexType BPPTREE_CONCAT(BPPTREE_MINMAX, _excluding)(IndexType begin, IndexType end, T exclude_index = Empty::empty) const {
37 IndexType min_max_index;
38 IndexType start;
39 if constexpr (enable_exclude) {
40 start = exclude_index == begin ? begin + 1 : begin;
41 } else {
42 start = begin;
43 }
44 if (start > end) {
45 min_max_index = -1;
46 } else {
47 min_max_index = start;
48 for (IndexType i = start + 1; i <= end; ++i) {
49 if constexpr (enable_exclude) {
50 if (i == exclude_index) {
51 continue;
52 }
53 }
54 if (comp(extractor(this->values[i]), extractor(this->values[min_max_index]))) {
55 min_max_index = i;
56 }
57 }
58 }
59 return min_max_index;

210 template <typename T = Empty, bool enable_exclude = !std::is_same_v<T, Empty>>
211 auto BPPTREE_CONCAT(BPPTREE_MINMAX, _excluding)(IndexType begin, IndexType end, T exclude_index = Empty::empty) const {
212 std::tuple<Key const*, IndexType> ret;
213 auto& [BPPTREE_MINMAX, min_max_index] = ret;
214 IndexType start;
215 if constexpr (enable_exclude) {
216 start = exclude_index == begin ? begin + 1 : begin;
217 } else {
218 start = begin;
219 }
220 if (start >= end) {
221 BPPTREE_MINMAX = nullptr;
222 min_max_index = -1;
223 } else {
224 BPPTREE_MINMAX = &BPPTREE_MINMAXS[start];
225 min_max_index = start;
226 for (IndexType i = start + 1; i < end; ++i) {
227 if constexpr (enable_exclude) {
228 if (i == exclude_index) {
229 continue;
230 }
231 }
232 if (comp(BPPTREE_MINMAXS[i], *BPPTREE_MINMAX)) {
233 BPPTREE_MINMAX = &BPPTREE_MINMAXS[i];
