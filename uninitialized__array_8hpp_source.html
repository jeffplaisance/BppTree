41template <typename T, size_t n>
42struct UninitializedArray {

44 alignas(T) std::array<std::byte, sizeof(T)*n> data;

48 UninitializedArray(UninitializedArray const& other, size_t length) noexcept {
49 if constexpr (std::is_trivially_copy_constructible_v<T>) {
50 std::memcpy(data.data(), other.data.data(), sizeof(T)*length);
51 } else {
52 T* base = reinterpret_cast<T*>(data.data());
53 T const* other_base = reinterpret_cast<T const*>(other.data.data());
54 for (size_t i = 0; i < length; ++i) {
55 new(base + i) T(other_base[i]);
56 }
57 }
