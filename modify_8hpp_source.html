20 static void collapse(TreeType& tree) {
21 while(tree.dispatch([](auto& tree, auto& root){
22 if constexpr (std::decay_t<decltype(root)>::Type::depth > 1) {

70 template <typename SplitType>
71 void operator()([[maybe_unused]] SplitType&& split) {
72 if constexpr (NodeType::depth < maxDepth) {
73 using NewRootType = InternalNode<NodeType::depth + 1>;
74 auto rootNode = makePtr<NewRootType>();
75 if (!split.left.ptrChanged) {
76 split.left.ptr = std::move(root);
77 split.left.ptrChanged = true;
78 }
79 rootNode->setElement(0, split.left);
80 rootNode->setElement(1, split.right);
81 rootNode->length = 2;
82 rootNode->setIndex(iter, split.new_element_left ? 0 : 1);
83 tree.rootVariant = std::move(rootNode);
84 } else {
85 throw std::logic_error("maximum depth exceeded");
86 }
87 }
