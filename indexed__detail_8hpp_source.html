177 start:
178 n = this->pointers[this->getIndex(it)]->advance(leaf, it, n);
179 if (n > 0) {
180 while (this->getIndex(it) < this->length - 1) {
181 this->incIndex(it);
182 if (static_cast<size_t>(childCounts[this->getIndex(it)]) < static_cast<size_t>(n)) {
183 n -= childCounts[this->getIndex(it)];
184 } else {
185 this->pointers[this->getIndex(it)]->seekFirst(it);
186 --n;
187 goto start;
188 }
189 }
190 return n;
191 }
192 if (n < 0) {
193 while (this->getIndex(it) > 0) {
194 this->decIndex(it);
195 if (static_cast<size_t>(childCounts[this->getIndex(it)]) < static_cast<size_t>(-n)) {
196 n += childCounts[this->getIndex(it)];
197 } else {
198 this->pointers[this->getIndex(it)]->seekLast(it);
199 ++n;
200 goto start;
201 }
