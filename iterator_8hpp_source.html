133  void advance(ssize n) {
134  n *= direction;
135  ssize remainder;
136  if (iter == rend) {
137  if (n <= 0) {
138  return;
139  }
140  iter = 0;
141  if (n == 1) {
142  std::as_const(*tree).dispatch([this](auto const& root) { root->seek_begin(leaf, iter); });
143  mod_count = tree->mod_count;
144  return;
145  }
146  remainder = n - 1;
147  } else if (mod_count == tree->mod_count) {
148  remainder = leaf->advance(leaf, iter, n);
149  } else {
150  remainder = n;
151  }
152  if (remainder != 0) {
153  std::as_const(*tree).dispatch([this, remainder](auto const& root) {
154  auto r = root->advance(leaf, iter, remainder);
155  if (r > 0) {
156  root->seek_end(leaf, iter);
157  }
158  if (r < 0) {
159  leaf = nullptr;
160  iter = rend;
161  }
162  });
163  mod_count = tree->mod_count;
164  }
165  }
