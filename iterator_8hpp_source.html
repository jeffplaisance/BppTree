39 [[nodiscard]] Value const& get() const {
40 if (mod_count != tree->mod_count) {
41 std::as_const(*tree).dispatch([this](auto const& root){
42 // make copy of iter since this is const and iter is not mutable.
43 // advancing by 0 won't actually change tmp_iter.
44 uint64_t tmp_iter = iter;
45 root->advance(leaf, tmp_iter, 0);
46 });
47 mod_count = tree->mod_count;
48 }
49 return leaf->getIter(iter);
50 }

132 void advance(ssize n) {
133 n *= direction;
134 ssize remainder;
135 if (iter == rend) {
136 if (n <= 0) {
137 return;
138 }
139 iter = 0;
140 if (n == 1) {
141 std::as_const(*tree).dispatch([this](auto const& root) { root->seekBegin(leaf, iter); });
142 mod_count = tree->mod_count;
143 return;
144 }
145 remainder = n - 1;
146 } else if (mod_count == tree->mod_count) {
147 remainder = leaf->advance(leaf, iter, n);
148 } else {
149 remainder = n;
150 }
151 if (remainder != 0) {
152 std::as_const(*tree).dispatch([this, remainder](auto const& root) {
153 auto r = root->advance(leaf, iter, remainder);
154 if (r > 0) {
155 root->seekEnd(leaf, iter);
156 }
157 if (r < 0) {
158 leaf = nullptr;
159 iter = rend;
160 }
161 });
162 mod_count = tree->mod_count;
163 }
164 }

215 [[nodiscard]] auto operator-(T const& it) const {
216 if constexpr (T::is_reversed) {
217 ssize ret = iter == rend ? 1 : -static_cast<ssize>(tree->order(*this));
218 ret += it.iter == rend ? -1 : static_cast<ssize>(tree->order(it));

220 } else {
