92 template <typename... Args>
93 void insertNoSplit(LeafNodeBase& node, IndexType index, Args&&... args) {
94 for (IndexType i = length - 1; i >= index; --i) {
95 if (persistent) {
96 node.values.set(i + 1, node.length, values[i]);
97 } else {
98 node.values.set(i + 1, node.length, values.move(i));
99 }
100 }
101 node.values.emplace(index, node.length, std::forward<Args>(args)...);
102 if (persistent) {
103 for (IndexType i = 0; i < index; ++i) {
104 node.values.set(i, node.length, values[i]);
105 }
106 }
107 node.length = length + 1;
108 }

119 template <typename... Args>
120 bool insertSplit(LeafNodeBase& left, LeafNodeBase& right, IndexType index, uint64_t& iter, bool rightMost, Args&&... args) {
121 IndexType split_point = rightMost && index == LeafSize ? index : (LeafSize + 1) / 2;
122 for (IndexType i = length - 1; i >= index; --i) {
123 if (persistent) {
124 setElement(left, right, i + 1, split_point, values[i]);
125 } else {
126 setElement(left, right, i + 1, split_point, std::move(values[i]));
127 }
128 }
129 if (index < split_point) {
130 left.values.emplace(index, left.length, std::forward<Args>(args)...);
131 } else {
132 right.values.emplace(index - split_point, right.length, std::forward<Args>(args)...);
133 }
134 for (IndexType i = persistent ? 0 : split_point; i < index; ++i) {
135 if (persistent) {
136 setElement(left, right, i, split_point, values[i]);
137 } else {
138 setElement(left, right, i, split_point, std::move(values[i]));
139 }
140 }
141 if constexpr (!std::is_trivially_destructible_v<Value>) {
142 for (IndexType i = split_point; i < left.length; ++i) {
143 left.values.destruct(i);
144 }
145 }
146 left.length = static_cast<uint16_t>(split_point);
147 right.length = static_cast<uint16_t>(LeafSize + 1 - split_point);
148 if (index >= split_point) {
149 setIndex(iter, index - split_point);
150 return false;
151 }
152 setIndex(iter, index);
153 return true;
154 }

156 template <typename R, typename S, typename... Args>
157 void insertIndex(IndexType index, R&& doReplace, S&& doSplit, size_t& size, uint64_t& iter, bool rightMost, Args&&... args) {
158 ++size;
159 if (length != LeafSize) {
160 setIndex(iter, index);
161 ReplaceType replace{};
162 computeDeltaInsert(index, replace.delta, args...);
163 if (persistent) {
164 replace.delta.ptr = makePtr<NodeType>();
165 replace.delta.ptrChanged = true;
166 insertNoSplit(*replace.delta.ptr, index, std::forward<Args>(args)...);
167 doReplace(replace);
168 } else {
169 insertNoSplit(this->self(), index, std::forward<Args>(args)...);
170 doReplace(replace);
171 }
172 } else {
173 auto right = makePtr<NodeType>();
174 if (persistent) {
175 NodePtr<NodeType> left = makePtr<NodeType>();
176 bool new_element_left = insertSplit(*left, *right, index, iter, rightMost, std::forward<Args>(args)...);
177 doSplit(SplitType(std::move(left), std::move(right), true, new_element_left));
178 } else {
179 bool new_element_left = insertSplit(this->self(), *right, index, iter, rightMost, std::forward<Args>(args)...);
180 doSplit(SplitType(&(this->self()), std::move(right), false, new_element_left));
181 }
182 }

213 bool erase(LeafNodeBase& node, IndexType index, uint64_t& iter, bool rightMost) {
214 if (persistent) {
215 for (IndexType i = 0; i < index; ++i) {
216 node.values.set(i, node.length, values[i]);
217 }
218 }
219 for (IndexType i = index + 1; i < length; ++i) {
220 if (persistent) {
221 node.values.set(i - 1, node.length, values[i]);
222 } else {
223 node.values.set(i - 1, node.length, values.move(i));
224 }
225 }
226 if (node.length == length) {
227 node.values.destruct(length - 1);
228 }
229 node.length = length - 1;
230 bool carry = index == node.length && !rightMost;
231 setIndex(iter, carry ? 0 : index);
232 return carry;
233 }

235 template <typename R, typename E>
236 void erase(IndexType index, R&& doReplace, E&& doErase, uint64_t& iter, bool rightMost) {
237 if (length > 1) {
238 ReplaceType replace{};
239 computeDeltaErase(index, replace.delta);
240 if (persistent) {
241 replace.delta.ptr = makePtr<NodeType>();
242 replace.delta.ptrChanged = true;
243 replace.carry = erase(*replace.delta.ptr, index, iter, rightMost);
244 doReplace(replace);
245 } else {
246 replace.carry = erase(this->self(), index, iter, rightMost);
247 doReplace(replace);
248 }
249 } else {
250 setIndex(iter, 0);
251 doErase();
252 }
253 }

316 ssize advance(typename LeafNodeBase::SelfType const*& leaf, uint64_t& it, ssize n) const {
317 auto sum = getIndex(it) + n;
318 if (sum >= length) {
319 auto ret = sum - (length - 1);
320 setIndex(it, length - 1);
321 return ret;
322 }
323 if (sum < 0) {
324 clearIndex(it);
325 return sum;
326 }
327 leaf = &this->self();
328 setIndex(it, sum);
