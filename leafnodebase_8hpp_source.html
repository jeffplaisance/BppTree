119  template <typename... Args>
120  bool insert_split(LeafNodeBase& left, LeafNodeBase& right, IndexType index, uint64_t& iter, bool right_most, Args&&... args) {
121  IndexType split_point = right_most && index == leaf_size ? index : (leaf_size + 1) / 2;
122  for (IndexType i = length - 1; i >= index; --i) {
123  if (persistent) {
124  set_element(left, right, i + 1, split_point, values[i]);
125  } else {
126  set_element(left, right, i + 1, split_point, std::move(values[i]));
127  }
128  }
129  if (index < split_point) {
130  left.values.emplace(index, left.length, std::forward<Args>(args)...);
131  } else {
132  right.values.emplace(index - split_point, right.length, std::forward<Args>(args)...);
133  }
134  for (IndexType i = persistent ? 0 : split_point; i < index; ++i) {
135  if (persistent) {
136  set_element(left, right, i, split_point, values[i]);
137  } else {
138  set_element(left, right, i, split_point, std::move(values[i]));
139  }
140  }
141  if constexpr (!std::is_trivially_destructible_v<Value>) {
142  for (IndexType i = split_point; i < left.length; ++i) {
143  left.values.destruct(i);
144  }
145  }
146  left.length = static_cast<uint16_t>(split_point);
147  right.length = static_cast<uint16_t>(leaf_size + 1 - split_point);
148  if (index >= split_point) {
149  set_index(iter, index - split_point);
150  return false;
151  }
152  set_index(iter, index);
153  return true;
154  }

156  template <typename R, typename S, typename... Args>
157  void insert_index(IndexType index, R&& do_replace, S&& do_split, size_t& size, uint64_t& iter, bool right_most, Args&&... args) {
158  ++size;
159  if (length != leaf_size) {
160  set_index(iter, index);
161  ReplaceType replace{};
162  compute_delta_insert(index, replace.delta, args...);
163  if (persistent) {
164  replace.delta.ptr = make_ptr<NodeType>();
165  replace.delta.ptr_changed = true;
166  insert_no_split(*replace.delta.ptr, index, std::forward<Args>(args)...);
167  do_replace(replace);
168  } else {
169  insert_no_split(this->self(), index, std::forward<Args>(args)...);
170  do_replace(replace);
171  }
172  } else {
173  auto right = make_ptr<NodeType>();
174  if (persistent) {
175  NodePtr<NodeType> left = make_ptr<NodeType>();
176  bool new_element_left = insert_split(*left, *right, index, iter, right_most, std::forward<Args>(args)...);
177  do_split(SplitType(std::move(left), std::move(right), true, new_element_left));
178  } else {
179  bool new_element_left = insert_split(this->self(), *right, index, iter, right_most, std::forward<Args>(args)...);
180  do_split(SplitType(&(this->self()), std::move(right), false, new_element_left));
181  }
182  }

213  bool erase(LeafNodeBase& node, IndexType index, uint64_t& iter, bool right_most) {
214  if (persistent) {
215  for (IndexType i = 0; i < index; ++i) {
216  node.values.set(i, node.length, values[i]);
217  }
218  }
219  for (IndexType i = index + 1; i < length; ++i) {
220  if (persistent) {
221  node.values.set(i - 1, node.length, values[i]);
222  } else {
223  node.values.set(i - 1, node.length, values.move(i));
224  }
225  }
226  if (node.length == length) {
227  node.values.destruct(length - 1);
228  }
229  node.length = length - 1;
230  bool carry = index == node.length && !right_most;
231  set_index(iter, carry ? 0 : index);
232  return carry;
233  }
