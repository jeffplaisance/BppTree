115  template <typename... Args>
116  bool insert_split(LeafNodeBase& left, LeafNodeBase& right, IndexType index, uint64_t& iter, bool right_most, Args&&... args) noexcept(disable_exceptions) {
117  IndexType split_point = right_most && index == leaf_size ? index : (leaf_size + 1) / 2;
118  for (IndexType i = length - 1; i >= index; --i) {
119  if (persistent) {
120  set_element(left, right, i + 1, split_point, values[i]);
121  } else {
122  set_element(left, right, i + 1, split_point, std::move(values[i]));
123  }
124  }
125  if (index < split_point) {
126  left.values.emplace(index, left.length, std::forward<Args>(args)...);
127  } else {
128  right.values.emplace(index - split_point, right.length, std::forward<Args>(args)...);
129  }
130  for (IndexType i = persistent ? 0 : split_point; i < index; ++i) {
131  if (persistent) {
132  set_element(left, right, i, split_point, values[i]);
133  } else {
134  set_element(left, right, i, split_point, std::move(values[i]));
135  }
136  }
137  if constexpr (!std::is_trivially_destructible_v<Value>) {
138  for (IndexType i = split_point; i < left.length; ++i) {
139  left.values.destruct(i);
140  }
141  }
142  left.length = static_cast<uint16_t>(split_point);
143  right.length = static_cast<uint16_t>(leaf_size + 1 - split_point);
144  if (index >= split_point) {
145  set_index(iter, index - split_point);
146  return false;
147  }
148  set_index(iter, index);
149  return true;
150  }

152  template <typename R, typename S, typename... Args>
153  void insert_index(IndexType index, R&& do_replace, S&& do_split, size_t& size, uint64_t& iter, bool right_most, Args&&... args) noexcept(disable_exceptions) {
154  ++size;
155  if (length != leaf_size) {
156  set_index(iter, index);
157  ReplaceType replace{};
158  compute_delta_insert(index, replace.delta, args...);
159  if (persistent) {
160  replace.delta.ptr = make_ptr<NodeType>();
161  replace.delta.ptr_changed = true;
162  insert_no_split(*replace.delta.ptr, index, std::forward<Args>(args)...);
163  do_replace(replace);
164  } else {
165  insert_no_split(this->self(), index, std::forward<Args>(args)...);
166  do_replace(replace);
167  }
168  } else {
169  auto right = make_ptr<NodeType>();
170  if (persistent) {
171  NodePtr<NodeType> left = make_ptr<NodeType>();
172  bool new_element_left = insert_split(*left, *right, index, iter, right_most, std::forward<Args>(args)...);
173  do_split(SplitType(std::move(left), std::move(right), true, new_element_left));
174  } else {
175  bool new_element_left = insert_split(this->self(), *right, index, iter, right_most, std::forward<Args>(args)...);
176  do_split(SplitType(&(this->self()), std::move(right), false, new_element_left));
177  }
178  }

209  bool erase(LeafNodeBase& node, IndexType index, uint64_t& iter, bool right_most) {
210  if (persistent) {
211  for (IndexType i = 0; i < index; ++i) {
212  node.values.set(i, node.length, values[i]);
213  }
214  }
215  for (IndexType i = index + 1; i < length; ++i) {
216  if (persistent) {
217  node.values.set(i - 1, node.length, values[i]);
218  } else {
219  node.values.set(i - 1, node.length, values.move(i));
220  }
221  }
222  if (node.length == length) {
223  node.values.destruct(length - 1);
224  }
225  node.length = length - 1;
226  bool carry = index == node.length && !right_most;
227  set_index(iter, carry ? 0 : index);
228  return carry;
