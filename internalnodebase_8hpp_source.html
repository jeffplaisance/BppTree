136 void insertReplace(IndexType index, ReplaceType<ChildType>& replace, R&& doReplace, uint64_t& iter) {
137 ReplaceType<NodeType> result{};
138 result.carry = replace.carry && index == length - 1;
139 setIndex(iter, !replace.carry ? index : result.carry ? 0 : index + 1);
140 computeDeltaReplace(replace.delta, result.delta, index);

143 result.delta.ptr->replaceElement(index, replace.delta);
144 result.delta.ptrChanged = true;
145 doReplace(result);
146 } else {

152 void insertSplitReplace(InternalNodeBase& node, IndexType index, SplitType<ChildType>& split) {
153 for (IndexType i = length - 1; i > index; --i) {
154 if (persistent) {

157 node.moveElement(i + 1, this->self(), i);

159 }
160 node.setElement(index, split.left);
161 node.setElement(index + 1, split.right);
162 if (persistent) {
163 for (IndexType i = 0; i < index; ++i) {
164 node.copyElement(i, this->self(), i);
165 }

200 bool insertSplitSplit(InternalNodeBase& left, InternalNodeBase& right, IndexType index, SplitType<ChildType>& split, uint64_t& iter, bool rightMost) {
201 IndexType split_point = rightMost && index + 1 == InternalSize ? index + 1 : (InternalSize + 1) / 2;
202 for (IndexType i = length - 1; i > index; --i) {
203 if (persistent) {
204 copyElement(left, right, i + 1, this->self(), i, split_point);
205 } else {
206 moveElement(left, right, i + 1, this->self(), i, split_point);
207 }
208 }
209 if (!split.left.ptrChanged) {
210 split.left.ptr = std::move(pointers[index]);
211 split.left.ptrChanged = true;
212 }
213 setElement(left, right, index, split_point, split.left);
214 setElement(left, right, index + 1, split_point, split.right);
215 for (IndexType i = persistent ? 0 : split_point; i < index; ++i) {
216 if (persistent) {
217 copyElement(left, right, i, this->self(), i, split_point);
218 } else {
219 moveElement(left, right, i, this->self(), i, split_point);
220 }
221 }
222 for (IndexType i = split_point; i < left.length; ++i) {
223 left.eraseElement(i);
224 }
225 left.length = static_cast<uint16_t>(split_point);
226 right.length = static_cast<uint16_t>(InternalSize + 1 - split_point);
227 IndexType insertion_index = split.new_element_left ? index : index + 1;
228 if (insertion_index >= split_point) {
229 setIndex(iter, insertion_index - split_point);
230 return false;
231 }
232 setIndex(iter, insertion_index);
233 return true;
234 }

236 template <typename R, typename S>
237 void insertSplit(
238 IndexType index,
239 SplitType<ChildType>& split,
240 R&& doReplace,
241 S&& doSplit,
242 uint64_t& iter,
243 bool rightMost
244 ) {
245 if (length != InternalSize) {
246 setIndex(iter, split.new_element_left ? index : index + 1);
247 ReplaceType<NodeType> replace{};
248 computeDeltaSplit(split, replace.delta, index);
249 if (persistent) {
250 replace.delta.ptr = makePtr<NodeType>();
251 insertSplitReplace(*replace.delta.ptr, index, split);
252 replace.delta.ptrChanged = true;
253 doReplace(replace);
254 } else {
255 insertSplitReplace(this->self(), index, split);
256 doReplace(replace);
257 }
258 } else {
259 auto right = makePtr<NodeType>();
260 if (persistent) {
261 auto left = makePtr<NodeType>();
262 bool new_element_left = insertSplitSplit(*left, *right, index, split, iter, rightMost);
263 doSplit(SplitType<NodeType>(std::move(left), std::move(right), true, new_element_left));
264 } else {
265 bool new_element_left = insertSplitSplit(this->self(), *right, index, split, iter, rightMost);
266 doSplit(SplitType<NodeType>(&(this->self()), std::move(right), false, new_element_left));
267 }
268 }

298 bool erase(InternalNodeBase& node, IndexType index, uint64_t& iter, bool rightMost) {
299 if (persistent) {
300 for (IndexType i = 0; i < index; ++i) {
301 node.copyElement(i, this->self(), i);
302 }
303 }
304 for (IndexType i = index + 1; i < length; ++i) {
305 if (persistent) {
306 node.copyElement(i - 1, this->self(), i);
307 } else {
308 node.moveElement(i - 1, this->self(), i);
309 }
310 }
311 if (node.length == length) {
312 node.eraseElement(length - 1);
313 }
314 node.length = length - 1;
315 bool carry = index == node.length;
316 if (carry && rightMost) {
317 node.seekEnd(iter);
318 return false;
319 }
320 setIndex(iter, carry ? 0 : index);
321 return carry;
322 }

428 ssize advance(typename InternalNodeBase::template LeafNodeType<LeafSize> const*& leaf, uint64_t& it, ssize n) const {
429 while (true) {
430 n = pointers[getIndex(it)]->advance(leaf, it, n);
431 if (n > 0) {
432 if (getIndex(it) < length - 1) {
433 incIndex(it);
434 pointers[getIndex(it)]->seekFirst(it);
435 --n;
436 continue;
437 }
438 return n;
439 }
440 if (n < 0) {
441 if (getIndex(it) > 0) {
442 decIndex(it);
443 pointers[getIndex(it)]->seekLast(it);
444 ++n;
445 continue;
446 }
447 return n;
448 }
449 return 0;
450 }
