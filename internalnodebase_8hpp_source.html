200  bool insert_split_split(InternalNodeBase& left, InternalNodeBase& right, IndexType index, SplitType<ChildType>& split, uint64_t& iter, bool right_most) {
201  IndexType split_point = right_most && index + 1 == internal_size ? index + 1 : (internal_size + 1) / 2;
202  for (IndexType i = length - 1; i > index; --i) {
203  if (persistent) {
204  copy_element(left, right, i + 1, this->self(), i, split_point);
205  } else {
206  move_element(left, right, i + 1, this->self(), i, split_point);
207  }
208  }
209  if (!split.left.ptr_changed) {
210  split.left.ptr = std::move(pointers[index]);
211  split.left.ptr_changed = true;
212  }
213  set_element(left, right, index, split_point, split.left);
214  set_element(left, right, index + 1, split_point, split.right);
215  for (IndexType i = persistent ? 0 : split_point; i < index; ++i) {
216  if (persistent) {
217  copy_element(left, right, i, this->self(), i, split_point);
218  } else {
219  move_element(left, right, i, this->self(), i, split_point);
220  }
221  }
222  for (IndexType i = split_point; i < left.length; ++i) {
223  left.erase_element(i);
224  }
225  left.length = static_cast<uint16_t>(split_point);
226  right.length = static_cast<uint16_t>(internal_size + 1 - split_point);
227  IndexType insertion_index = split.new_element_left ? index : index + 1;
228  if (insertion_index >= split_point) {
229  set_index(iter, insertion_index - split_point);
230  return false;
231  }
232  set_index(iter, insertion_index);
233  return true;
234  }

298  bool erase(InternalNodeBase& node, IndexType index, uint64_t& iter, bool right_most) {
299  if (persistent) {
300  for (IndexType i = 0; i < index; ++i) {
301  node.copy_element(i, this->self(), i);
302  }
303  }
304  for (IndexType i = index + 1; i < length; ++i) {
305  if (persistent) {
306  node.copy_element(i - 1, this->self(), i);
307  } else {
308  node.move_element(i - 1, this->self(), i);
309  }
310  }
311  if (node.length == length) {
312  node.erase_element(length - 1);
313  }
314  node.length = length - 1;
315  bool carry = index == node.length;
316  if (carry && right_most) {
317  node.seek_end(iter);
318  return false;
319  }
320  set_index(iter, carry ? 0 : index);
321  return carry;
